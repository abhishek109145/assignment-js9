        Ques: "Problem:
            You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)
            
            Input:
            abcadeecfb
            
            Output:
            a=2
            b=2
            c=2
            d=1
            e=2
            f=1"
            Ans:-
            
            let str = "abcadeecfb";
            let map = new Map();
            
            let len = str.length;
            for(let i=0;i< len;i++)
            {
    
                let char = str.charAt(i);
                if(map.has(char))
                {
                    let value = map.get(char);
                    map.set(char,value+1);
                }
                else
                {
                    map.set(char,1);
                }
            }
            console.log(map);
